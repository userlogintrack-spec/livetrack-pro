window.onerror = function(msg, source, line, col, error) {
  fetch('/api/track/js-error/', {
    method: 'POST',
    headers: {'Content-Type': 'application/json'},
    credentials: 'include',
    body: JSON.stringify({
      message: msg,
      source: source,
      line: line, col: col,
      stack: error ? error.stack : '',
      url: window.location.href
    })
  });
};
