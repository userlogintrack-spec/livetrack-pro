Event Tracking Code

Add this JavaScript to your website to track custom events:

// Track a custom event
fetch('/api/track/event/', {
  method: 'POST',
  headers: {'Content-Type': 'application/json'},
  credentials: 'include',
  body: JSON.stringify({
    name: 'button_click',
    category: 'engagement',
    label: 'Sign Up Button',
    value: 1,
    page_url: window.location.href
  })
});

// Track page performance
fetch('/api/track/performance/', {
  method: 'POST',
  headers: {'Content-Type': 'application/json'},
  credentials: 'include',
  body: JSON.stringify({
    url: window.location.pathname,
    load_time_ms: performance.timing.loadEventEnd - performance.timing.navigationStart,
    time_spent: 30
  })
});
